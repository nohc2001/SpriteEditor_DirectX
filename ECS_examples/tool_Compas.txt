pEventSystem evtsys;
pPostRenderTarget prt;
pICBEditor selectedICBE;

int main(){
	EventData evt;
	pRenderData rd0 = 0;

	while(true){
		while(true){
			evt = PopEvent(evtsys);
			if(evt.message != 0){
				break;
			}
		}

		//event loop
		if(KeyDown(evt)){
			if(getEventKey(evt) == 'A'){
				ChangeRenderData(&rd0, _rdCircle(0, 0, 100, _gcolor(1, 1, 1, 1)), 0);
				ReleaseRenderData(0);
			}
		}
	}
}
